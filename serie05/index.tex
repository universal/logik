  & \{X_2, X_4\} & (A)\\

  & \{\neg X_2\} & (A)\\
  & \{X_5\} & (R) & auf 7 + 8\\
  & \{\neg X_5\} & (A)\\
  & \{\} & (R) & auf 9 + 10
\end{align}
